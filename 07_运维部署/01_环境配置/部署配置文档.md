## 3. Docker Compose 配置

```yaml
version: '3.8'

services:
  nginx:
    image: nginx:latest
    container_name: eduadmin-nginx
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./nginx/ssl:/etc/nginx/ssl
      - ./frontend/dist:/usr/share/nginx/html
      - ./logs/nginx:/var/log/nginx
    depends_on:
      - backend
    restart: always

  backend:
    image: eduadmin-backend:latest
    container_name: eduadmin-backend
    environment:
      - DEBUG=False
      - DB_HOST=postgres
      - DB_PORT=5432
      - DB_NAME=eduadmin
      - DB_USER=postgres
      - DB_PASSWORD=${DB_PASSWORD}
      - REDIS_URL=redis://redis:6379/1
      - JWT_SECRET_KEY=${JWT_SECRET_KEY}
    volumes:
      - ./backend/media:/code/media
      - ./logs:/code/logs
    depends_on:
      - postgres
      - redis
    restart: always

  postgres:
    image: postgres:14
    container_name: eduadmin-postgres
    environment:
      - POSTGRES_DB=eduadmin
      - POSTGRES_USER=postgres
      - POSTGRES_PASSWORD=${DB_PASSWORD}
    volumes:
      - ./data/postgres:/var/lib/postgresql/data
    restart: always

  redis:
    image: redis:7-alpine
    container_name: eduadmin-redis
    volumes:
      - ./data/redis:/data
    restart: always

  celery:
    image: eduadmin-backend:latest
    container_name: eduadmin-celery
    command: celery -A eduadmin worker -l info
    environment:
      - CELERY_BROKER_URL=redis://redis:6379/0
      - CELERY_RESULT_BACKEND=redis://redis:6379/0
    depends_on:
      - redis
    restart: always
```

## 4. Nginx 配置

```nginx
upstream eduadmin_backend {
    server backend:8000;
}

server {
    listen 80;
    server_name edu.example.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name edu.example.com;

    ssl_certificate /etc/nginx/ssl/fullchain.pem;
    ssl_certificate_key /etc/nginx/ssl/privkey.pem;
    ssl_session_timeout 1d;
    ssl_session_cache shared:SSL:50m;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers ECDHE-ECDSA-AES128-GCM-SHA256:ECDHE-RSA-AES128-GCM-SHA256;

    # 前端静态文件
    location / {
        root /usr/share/nginx/html;
        index index.html;
        try_files $uri $uri/ /index.html;
    }

    # 后端API
    location /api {
        proxy_pass http://eduadmin_backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # 媒体文件
    location /media {
        alias /code/media;
        expires 30d;
    }

    # 静态文件
    location /static {
        alias /code/staticfiles;
        expires 30d;
    }

    # WebSocket
    location /ws {
        proxy_pass http://eduadmin_backend;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
    }
}
```

## 6. 部署步骤

### 6.1 拉取代码

```bash
git clone https://github.com/your-org/eduadmin.git
cd eduadmin
```

### 6.2 配置环境变量

```bash
cp .env.example .env
# 编辑 .env 文件
```

### 6.3 启动服务

```bash
docker-compose up -d
```

### 6.4 数据库迁移

```bash
docker-compose exec backend python manage.py migrate
docker-compose exec backend python manage.py createsuperuser
```

### 6.5 收集静态文件

```bash
docker-compose exec backend python manage.py collectstatic --noinput
```

### 6.6 重启服务

```bash
docker-compose restart
```

## 8. 备份策略

### 8.1 数据库备份

每天凌晨2点执行：

```bash
docker-compose exec postgres pg_dump -U postgres eduadmin > backup_$(date +%Y%m%d).sql
```

### 8.2 文件备份

媒体文件和配置每周备份：

```bash
tar -czf backup_media_$(date +%Y%m%d).tar.gz ./backend/media
```

## 9. 故障恢复

### 9.1 数据库恢复

```bash
docker-compose exec -T postgres psql -U postgres eduadmin < backup_20240115.sql
```

### 9.2 回滚版本

```bash
git checkout v1.0.0
docker-compose build backend
docker-compose up -d backend
```
